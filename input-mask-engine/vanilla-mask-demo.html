Vanilla JS Mask Demo

Supports both data-mask and data-mask-free using your existing input-mask engine.

Use

<script src="./input-mask-engine.js"></script>
<script src="./vanilla-adapter.js"></script>

<input data-mask="(999) 999 - 9999">
<input data-mask-free="{ token: '9', max: 6 }">

<script>
MaskEngineDOM.init({
  tokens: {
    Z: {
      test: ch => /[A-Fa-f0-9]/.test(ch),
      transform: ch => ch.toUpperCase()
    }
  }
});
</script>
